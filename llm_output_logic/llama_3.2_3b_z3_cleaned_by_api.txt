```python
import z3

# Define the variables
movie_cost = 2000
dvd_cost = 6
selling_price = movie_cost * 2.5
daily_sales = 500
days_per_week = 5
weeks = 20

# Create a Z3 solver
solver = z3.Solver()

# Calculate daily revenue
daily_revenue = selling_price * daily_sales
weekly_revenue = daily_revenue * days_per_week

# Calculate weekly cost
weekly_cost = dvd_cost * daily_sales + movie_cost

# Add the weekly revenue and cost to the solver
solver.add(weekly_revenue - weekly_cost >= 0)

# Solve the problem
result = solver.check()

if result == z3.sat:
    # Get the maximum profit
    max_profit = weekly_revenue - weekly_cost
    print(f"Maximum Profit: {max_profit}")
else:
    print("No solution found")
```
