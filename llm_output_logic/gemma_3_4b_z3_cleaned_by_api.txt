def solve():
    # Let s be the number of letters in Samantha's last name.
    # Let b be the number of letters in Bobbie's last name.
    # Let j be the number of letters in Jamie's last name (Grey).

    # We are given that Samantha's last name has three fewer letters than Bobbie's last name:
    # s = b - 3

    # If Bobbie took two letters off her last name, she would have a last name twice the length of Jamie's:
    # b - 2 = 2 * (j)

    # Jamie's full name is Jamie Grey. So, j = 5 (Jamie) + 4 (Grey) = 9

    # Substitute j = 9 into the second equation:
    # b - 2 = 2 * 9
    # b - 2 = 18
    # b = 20

    # Now substitute b = 20 into the first equation:
    # s = 20 - 3 = 17

    print("Samantha's last name has", 17, "letters")
    print("Bobbie's last name has", 20, "letters")
    print("Jamie's last name has", 9, "letters")

solve()
