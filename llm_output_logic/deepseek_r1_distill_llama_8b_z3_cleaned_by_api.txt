== Problem 64 ===


```python
from z3 import *

# Define the variables
cost_per_top = Real('cost_per_top')

# Define the known values
shorts_count = 5
shorts_cost_per = 7
shoes_count = 2
shoes_cost_per = 10
tops_count = 4

# Calculate total cost from shorts and shoes
total_shorts_cost = shorts_count * shorts_cost_per
total_shoes_cost = shoes_count * shoes_cost_per
total_known_cost = total_shorts_cost + total_shoes_cost

# The total cost from tops should equal the total known cost
s = Solver()
s.add(tops_count * cost_per_top == total_known_cost)

# Check if there's a solution
if s.check() == sat:
    m = s.model()
    result = m[cost_per_top].as_decimal(2)
    print(result)
else:
    print("No solution found")
```

== Problem 76 ===


```python
from z3 import Int, solve

# Define variables
total_pages = Int('total_pages')
yesterday_read = Int('yesterday_read')
today_read = Int('today_read')
remaining_after_yesterday = Int('remaining_after_yesterday')
remaining_after_today = Int('remaining_after_today')
tomorrow_read = Int('tomorrow_read')

# Constraints
constraints = [
    total_pages == 120,
    yesterday_read == 12,
    remaining_after_yesterday == total_pages - yesterday_read,
    today_read == 2 * yesterday_read,
    remaining_after_today == remaining_after_yesterday - today_read,
    tomorrow_read == remaining_after_today / 2
]

# Solve
solve(constraints)
```
