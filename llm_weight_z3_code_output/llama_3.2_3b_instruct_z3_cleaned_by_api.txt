=== Cleaning Problem 31 ===


```python
from sympy import symbols, Eq, solve

# Define variables
oil_volume = symbols('oil_volume')
onion_weight = symbols('onion_weight')

# Given values
oil_density = 920  # g/l
pan_contents_volume = 750  # ml
chopped_onions_weight = 300  # g

# Convert oil volume from ml to l
oil_volume_l = pan_contents_volume / 1000

# Calculate weight of oil in the pan
oil_weight = oil_density * oil_volume_l

# Add weight of chopped onions
total_weight = Eq(oil_weight + onion_weight, symbols('total_weight'))

# Solve for total weight
solution = solve(total_weight, symbols('total_weight'))
print(solution[0])
```
