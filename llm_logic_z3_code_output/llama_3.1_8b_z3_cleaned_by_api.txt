== Problem 88 ===


```python
from z3 import *

hourly_wage = 18
overtime_rate = 0.5
hours_per_shift = 8
total_hours_worked = 50
shifts_per_week = 5

regular_earnings = (hours_per_shift * hourly_wage) * shifts_per_week
overtime_hours = total_hours_worked - (hours_per_shift * shifts_per_week)

if overtime_hours > 0:
    overtime_pay = (hourly_wage + (hourly_wage * overtime_rate)) * overtime_hours
else:
    overtime_pay = 0

total_earnings = regular_earnings + overtime_pay
print("Total Earnings: $", int(total_earnings))
```
