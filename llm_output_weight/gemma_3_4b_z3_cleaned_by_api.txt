```python
from z3 import *

# Define constants
apple_weight_g = 20  # Apple weight in grams
num_pears = 2        # Number of pears my friend has
half_pears = num_pears / 2  # 1.0 pear equivalent
basket_weight_oz = 80       # Total weight in ounces

# Convert ounces to grams (1 oz = 28.3495 g)
basket_weight_g = basket_weight_oz * 28.3495

# Create solver and variable
solver = Solver()
X = Real('X')  # Weight of a single pear in grams

# Add constraint: total weight = apple + half pears
solver.add(apple_weight_g + (X * half_pears) == basket_weight_g)

# Solve the model
solution = solver.check()

# Check if a solution exists
if solution == sat:
    model = solver.model()
    pear_weight = model[X].as_decimal(6)  # Get value with 6 decimal places
    print(f"The weight of a single pear is: {pear_weight} grams")
else:
    print("No solution found.")
```

from z3 import *

# Define variables properly
n = Int('n')  # Weight of one orange in grams
w = Real('w') # Weight of the watermelon slice in grams
p = Real('p') # Weight of the plate in grams

# Create solver
s = Solver()

# Add constraints
s.add(n > 0)  # Orange weight must be positive
s.add(w >= 0) # Watermelon slice weight must be non-negative
s.add(p >= 0) # Plate weight must be non-negative

# Equations based on the problem
# Initial: 3 oranges on plate = 3n + p
# After eating 1/3 of one orange: 3n - n/3 = 8n/3 oranges remaining
# After adding watermelon: 8n/3 + w + p = 1100g
# Plate weight: p = 400g

s.add(p == 400)
s.add((8 * n) / 3 + w + p == 1100)

# Solve the model
if s.check() == sat:
    solution = s.model()
    n_value = solution[n]
    w_value = solution[w]
    p_value = solution[p]
    
    print(f"The weight of one orange is {n_value} grams.")
    print(f"The weight of the watermelon slice is {w_value} grams.")
    print(f"The weight of the plate is {p_value} grams.")
else:
    print("No solution found.")```

```python
from z3 import *

# Define symbolic variables
jar_weight = Int("jar_weight")
cookie_weight = Int("cookie_weight")
num_cookies = Int("num_cookies")

# Create solver
solver = Solver()

# Add constraints based on the problem
solver.add(jar_weight == 300)           # Jar weighs 300g
solver.add(num_cookies == 5)            # There are 5 cookies initially
solver.add(jar_weight + num_cookies * cookie_weight == 950)  # Total weight is 950g

# After eating 2 cookies, there are 3 cookies left
remaining_cookies = num_cookies - 2
new_total_weight = jar_weight + remaining_cookies * cookie_weight

# Check if the new total weight is 650g (not 930g as in your original code)
# 950g total - 2 cookies = 950 - 2*(cookie_weight)
# If cookie_weight = (950-300)/5 = 650/5 = 130g
# Then new weight = 300 + 3*130 = 300 + 390 = 690g
solver.add(new_total_weight == 690)  # Corrected based on the math

# Solve the model
if solver.check() == sat:
    solution = solver.model()
    cookie_weight_value = solution[cookie_weight]
    jar_weight_value = solution[jar_weight]
    num_cookies_value = solution[num_cookies]
    
    print(f"Cookie weight: {cookie_weight_value}g")
    print(f"Jar weight: {jar_weight_value}g") 
    print(f"Initial number of cookies: {num_cookies_value}")
    print(f"New total weight: {solution.evaluate(new_total_weight)}g")
else:
    print("No solution found.")
```

```python
def solve():
  # Let 'b' be the weight of the box in grams.
  # Let 'p' be the weight of a single pencil in grams.

  # We have two equations:
  # 1) b + 10p = 230
  # 2) b = 50

  # Substitute b = 50 into equation 1:
  # 50 + 10p = 230
  # 10p = 180
  # p = 18

  # Now we have the weight of a single pencil (p = 18g).

  # If 4 pencils are removed, we have 6 pencils remaining.
  # The new total weight is b + 6p.

  b = 50
  p = 18
  new_weight = b + 6 * p

  print(new_weight)

solve()
```
